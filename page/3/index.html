Haproxy 转发

|

/etc/haproxy/haproxy.cfg

global
        ulimit-n  51200

defaults
        log global
        mode    tcp
        option  dontlognull
        contimeout 1000
        clitimeout 150000
        srvtimeout 150000


frontend ss-in
        bind *:7890
        default_backend ss-out
backend ss-out
        server server1 xx.xx.xx.xx:1026 maxconn 20480

frontend ss-in1
         bind *:7891
         default_backend ss-out1
backend ss-out1
        server server3 xx.xx.xx.xx:1030 maxconn 20480

frontend ss-in2
           bind *:7892
           default_backend ss-out2
 backend ss-out2
          server server3 xx.xx.xx.xx:1028 maxconn 20480
